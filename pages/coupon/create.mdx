## Criar Cupom

Cria um novo cupom de desconto.

<CodeGroup>

```bash cURL
curl -X POST "https://api.upay-sistema.onrender.com/api/coupons" \
  -H "Authorization: Bearer SUA_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "code": "DESCONTO10",
    "discountType": "PERCENTAGE",
    "discountValue": 10,
    "description": "10% de desconto"
  }'
```

```javascript JavaScript
const response = await fetch('https://api.upay-sistema.onrender.com/api/coupons', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer SUA_API_KEY',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    code: 'DESCONTO10',
    discountType: 'PERCENTAGE',
    discountValue: 10,
    description: '10% de desconto'
  })
});

const data = await response.json();
```

```python Python
import requests

url = "https://api.upay-sistema.onrender.com/api/coupons"
headers = {
    "Authorization": "Bearer SUA_API_KEY",
    "Content-Type": "application/json"
}
payload = {
    "code": "DESCONTO10",
    "discountType": "PERCENTAGE",
    "discountValue": 10,
    "description": "10% de desconto"
}

response = requests.post(url, json=payload, headers=headers)
data = response.json()
```

</CodeGroup>

<ParamField body="code" type="string" required>
  Código do cupom (único, máx. 50 caracteres)
</ParamField>
<ParamField body="discountType" type="enum" required>
  "PERCENTAGE" ou "FIXED"
</ParamField>
<ParamField body="discountValue" type="number" required>
  Para PERCENTAGE: valor percentual (ex: 10 = 10%)
  Para FIXED: valor em centavos (ex: 5000 = R$ 50,00)
</ParamField>
<ParamField body="description" type="string">
  Descrição do cupom
</ParamField>
<ParamField body="minPurchaseAmount" type="number">
  Valor mínimo de compra em centavos
</ParamField>
<ParamField body="maxDiscountAmount" type="number">
  Valor máximo de desconto em centavos (apenas para PERCENTAGE)
</ParamField>
<ParamField body="maxUses" type="number">
  Limite total de usos (null = ilimitado)
</ParamField>
<ParamField body="maxUsesPerUser" type="number">
  Limite de usos por usuário (null = ilimitado)
</ParamField>
<ParamField body="validFrom" type="string">
  Data de início (ISO 8601)
</ParamField>
<ParamField body="validUntil" type="string">
  Data de fim (ISO 8601)
</ParamField>
<ParamField body="productIds" type="array">
  IDs dos produtos aplicáveis (vazio = todos)
</ParamField>
<ParamField body="status" type="enum">
  "ACTIVE" ou "INACTIVE" (padrão: "ACTIVE")
</ParamField>
