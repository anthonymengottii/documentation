## Criar Produto

Cria um novo produto no seu catálogo.

<CodeGroup>

```bash cURL
curl -X POST "https://api.upay-sistema.onrender.com/api/products" \
  -H "Authorization: Bearer SUA_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "name": "Curso de JavaScript",
    "description": "Curso completo de JavaScript",
    "priceCents": 19900,
    "currency": "BRL",
    "stockQuantity": 100,
    "stockEnabled": true
  }'
```

```javascript JavaScript
const response = await fetch('https://api.upay-sistema.onrender.com/api/products', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer SUA_API_KEY',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    name: 'Curso de JavaScript',
    description: 'Curso completo de JavaScript',
    priceCents: 19900,
    currency: 'BRL',
    stockQuantity: 100,
    stockEnabled: true
  })
});

const data = await response.json();
```

```python Python
import requests

url = "https://api.upay-sistema.onrender.com/api/products"
headers = {
    "Authorization": "Bearer SUA_API_KEY",
    "Content-Type": "application/json"
}
payload = {
    "name": "Curso de JavaScript",
    "description": "Curso completo de JavaScript",
    "priceCents": 19900,
    "currency": "BRL",
    "stockQuantity": 100,
    "stockEnabled": True
}

response = requests.post(url, json=payload, headers=headers)
data = response.json()
```

</CodeGroup>

<ParamField body="name" type="string" required>
  Nome do produto (máx. 255 caracteres)
</ParamField>
<ParamField body="description" type="string">
  Descrição detalhada do produto
</ParamField>
<ParamField body="priceCents" type="number">
  Preço em centavos (ex: 19900 = R$ 199,00)
</ParamField>
<ParamField body="currency" type="string">
  Moeda (padrão: "BRL")
</ParamField>
<ParamField body="stockQuantity" type="number">
  Quantidade em estoque
</ParamField>
<ParamField body="stockEnabled" type="boolean">
  Ativar controle de estoque (padrão: false)
</ParamField>
<ParamField body="status" type="enum">
  "ACTIVE" ou "INACTIVE" (padrão: "ACTIVE")
</ParamField>

<ResponseField name="message" type="string">
  Mensagem de sucesso
</ResponseField>
<ResponseField name="data" type="object">
  Dados do produto criado
</ResponseField>
<ResponseField name="data.id" type="string">
  ID único do produto
</ResponseField>
<ResponseField name="data.name" type="string">
  Nome do produto
</ResponseField>
<ResponseField name="data.priceCents" type="number">
  Preço em centavos
</ResponseField>
<ResponseField name="data.stockQuantity" type="number">
  Quantidade em estoque
</ResponseField>
<ResponseField name="data.status" type="string">
  Status do produto: "ACTIVE" ou "INACTIVE"
</ResponseField>
