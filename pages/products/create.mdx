## Criar Produto

Cria um novo produto no seu catálogo.

<RequestExample>

```bash
curl -X POST "https://api.upay-sistema.onrender.com/api/products" \
  -H "Authorization: Bearer SUA_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "name": "Curso de JavaScript",
    "description": "Curso completo de JavaScript",
    "priceCents": 19900,
    "currency": "BRL",
    "stockQuantity": 100,
    "stockEnabled": true
  }'
```

</RequestExample>

<ParamField body="name" type="string" required>
  Nome do produto (máx. 255 caracteres)
</ParamField>
<ParamField body="description" type="string">
  Descrição detalhada do produto
</ParamField>
<ParamField body="priceCents" type="number">
  Preço em centavos (ex: 19900 = R$ 199,00)
</ParamField>
<ParamField body="currency" type="string">
  Moeda (padrão: "BRL")
</ParamField>
<ParamField body="stockQuantity" type="number">
  Quantidade em estoque
</ParamField>
<ParamField body="stockEnabled" type="boolean">
  Ativar controle de estoque (padrão: false)
</ParamField>
<ParamField body="status" type="enum">
  "ACTIVE" ou "INACTIVE" (padrão: "ACTIVE")
</ParamField>

<ResponseField name="success" type="boolean">
  Indica se a requisição foi bem-sucedida
</ResponseField>
<ResponseField name="data" type="object">
  Dados do produto criado
</ResponseField>
