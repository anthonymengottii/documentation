## Listar Produtos

Retorna uma lista paginada de todos os produtos do catálogo.

<CodeGroup>

```bash cURL
curl -X GET "https://api.upay-sistema.onrender.com/api/products?page=1&limit=20" \
  -H "Authorization: Bearer SUA_API_KEY"
```

```javascript JavaScript
const response = await fetch('https://api.upay-sistema.onrender.com/api/products?page=1&limit=20', {
  method: 'GET',
  headers: {
    'Authorization': 'Bearer SUA_API_KEY'
  }
});

const data = await response.json();
```

```python Python
import requests

url = "https://api.upay-sistema.onrender.com/api/products"
headers = {
    "Authorization": "Bearer SUA_API_KEY"
}
params = {
    "page": 1,
    "limit": 20
}

response = requests.get(url, params=params, headers=headers)
data = response.json()
```

</CodeGroup>

<ParamField query="page" type="number">
  Número da página (padrão: 1)
</ParamField>
<ParamField query="limit" type="number">
  Itens por página (padrão: 20, máx: 100)
</ParamField>
<ParamField query="status" type="enum">
  Filtrar por status: "ACTIVE" ou "INACTIVE"
</ParamField>

<ResponseField name="data" type="array">
  Lista de produtos
</ResponseField>
<ResponseField name="pagination" type="object">
  Informações de paginação
</ResponseField>
<ResponseField name="pagination.page" type="number">
  Página atual
</ResponseField>
<ResponseField name="pagination.limit" type="number">
  Itens por página
</ResponseField>
<ResponseField name="pagination.total" type="number">
  Total de itens
</ResponseField>
